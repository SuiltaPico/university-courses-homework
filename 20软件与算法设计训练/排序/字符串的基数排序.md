# 基数排序（字符串）
原理是把字符串当做`char`数组计算。
## 代码展示
```cpp
#include <vector>
#include <cmath>
#include <cstring>
#include <cstdio>
using namespace std;
void radixSort(char* _str)
{
    size_t arr_len = strlen(_str);
    vector<char> stacks[10];
    for (size_t i = 0; i < 10; i++)
        stacks[i] = vector<char>();
    char max = _str[0];
    int max_power_of_10 = 0;
    for (size_t i = 1; i < arr_len; i++)
        if (_str[i] > max)
            max = _str[i];
    while (max / (int)pow(10, max_power_of_10) != 0)
        max_power_of_10++;
    for (int power_of_10 = 0; power_of_10 < max_power_of_10; power_of_10++)
    {
        for (size_t i = 0; i < arr_len; i++)
            stacks[_str[i] / (int)pow(10, power_of_10) % 10].push_back(_str[i]);
        size_t arr_index = 0;
        for (size_t i = 0; i < 10; i++)
        {
            for (size_t j = 0; j < stacks[i].size(); j++)
            {
                _str[arr_index] = stacks[i].at(j);
                arr_index++;
            }
            stacks[i].clear();
        }
    }
}
int main()
{
    char _str[] = "uiahygrab";
    radixSort(_str);
    printf("%s", _str);
    return 0;
}
```
