# 二分查找变体
## 代码展示
```cpp
#include <cstdio>

// 第一个等于给定值的元素
int binary_search_first(int *arr, size_t arr_size, int target)
{
    size_t left = 0, right = arr_size - 1;
    while (left < right)
    {
        size_t mid = (left + right) / 2;
        if (target > arr[mid])
            left = mid + 1;
        else if (target < arr[mid] || (mid != 0 && target == arr[mid - 1]))
            right = mid - 1;
        else
            return mid;
    }

    if (target == arr[left])
        return left;

    return -1;
}

// 最后一个等于给定值的元素
int binary_search_last(int *arr, size_t arr_size, int target)
{
    size_t left = 0, right = arr_size - 1;
    while (left < right)
    {
        size_t mid = (left + right) / 2;
        if (target > arr[mid] || (mid != arr_size - 1 && target == arr[mid + 1]))
            left = mid + 1;
        else if (target < arr[mid])
            right = mid - 1;
        else
            return mid;
    }

    if (target == arr[left])
        return left;

    return -1;
}

// 第一个大于给定值的元素
int binary_search_first_greater_than_this(int *arr, size_t arr_size, int target)
{
    int last_index = binary_search_last(arr, arr_size, target);
    if (last_index > -1 && last_index + 1 < arr_size)
        return last_index + 1;
    return -1;
}

// 第一个小于给定值的元素
int binary_search_first_less_than_this(int *arr, size_t arr_size, int target)
{
    int first_index = binary_search_first(arr, arr_size, target);
    if (first_index > 0)
        return first_index - 1;
    return -1;
}

int main()
{
    int arr[] = {1, 1, 2, 2, 2, 3, 5, 6, 9, 9, 10, 15};
    printf("first 0: %d\n", binary_search_first(arr, sizeof(arr) / sizeof(int), 0));
    printf("first 2: %d\n", binary_search_first(arr, sizeof(arr) / sizeof(int), 2));
    printf("first greater than 2: %d\n", binary_search_last(arr, sizeof(arr) / sizeof(int), 2));
    printf("first less than 2: %d\n", binary_search_last(arr, sizeof(arr) / sizeof(int), 2));
    return 0;
}
```
